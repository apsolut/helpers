    %}

{% dnd_column width=4 %}
    {% dnd_row%}

        {% dnd_module "image" path="@hubspot/image",flexbox_positioning='MIDDLE_CENTTER', label="Featured Image" %}

    {% end_dnd_row%}
{% end_dnd_column %}

{% dnd_column width=8, offset=4 %}
    {% dnd_row%}

        {% dnd_module "rich_text" path="@hubspot/rich_text",label="Featured Text" %}
        {% end_dnd_module %}
    
    {% end_dnd_row%}
{% end_dnd_column %}

{% end_dnd_section %}

{% end_dnd_area %}
